-- Check lương nhân viên
CREATE TRIGGER CHECKLUONGNV ON [NHAN_VIEN]
AFTER INSERT, UPDATE
AS 
BEGIN
	DECLARE @LUONG MONEY
	SELECT @LUONG = mucluong FROM inserted 

	IF @LUONG > ANY (SELECT mucluong FROM NHAN_VIEN WHERE mavaitro = 'CV01')
	BEGIN
		PRINT N'Mức lương nhân viên không được lớn hơn hoặc bằng mức lương quản lý'
		ROLLBACK
	END
END

CREATE TRIGGER CheckNgayDatHang
ON PHIEU_DAT_HANG
AFTER UPDATE
AS
BEGIN
    DECLARE @mapd NVARCHAR(50);
    DECLARE @ngaydat DATE;

    -- Lấy giá trị mapd và ngaydat từ bản ghi vừa chèn hoặc cập nhật
    SELECT @mapd = mapd, @ngaydat = ngaydat
    FROM inserted;

    -- Kiểm tra điều kiện
    IF EXISTS (
        SELECT 1
        FROM PHIEU_NHAP_HANG
        WHERE mapd = @mapd AND ngaynhap < @ngaydat
    )
    BEGIN
        -- Nếu có bản ghi không hợp lệ, rollback transaction
        PRINT N'Ngày đặt không được lớn hơn ngày nhập'
        ROLLBACK TRANSACTION;
    END
END;



CREATE TRIGGER CheckNgayNhapHang
ON PHIEU_NHAP_HANG
AFTER INSERT, UPDATE
AS
BEGIN
    DECLARE @mapd NVARCHAR(50);
    DECLARE @ngaynhap DATE;

    SELECT @mapd = mapd, @ngaynhap = ngaynhap
    FROM inserted;

    -- Kiểm tra điều kiện cho tất cả các bản ghi trong bảng inserted
    IF EXISTS (
        SELECT 1
        FROM PHIEU_DAT_HANG d 
        WHERE @ngaynhap < d.ngaydat AND mapd = @mapd
    )
    BEGIN
        -- Nếu có bản ghi không hợp lệ, rollback transaction
        PRINT N'Ngày nhập không được nhỏ hơn ngày đặt'
        ROLLBACK TRANSACTION;
    END
END;




CREATE TRIGGER CheckSoluongdat
ON CHI_TIET_PHIEU_DAT
AFTER UPDATE
AS
BEGIN
    DECLARE @mapd CHAR(20);
    DECLARE @masp CHAR(10);
    DECLARE @soluongdat INT;
    DECLARE @tongsoluongnhap INT;

    -- Lấy thông tin từ bảng INSERTED
    SELECT @mapd = mapd, @masp = masp
    FROM INSERTED;

    -- Lấy số lượng đã đặt cho sản phẩm trong bảng CTPD
    SELECT @soluongdat = soluongdat
    FROM CHI_TIET_PHIEU_DAT
    WHERE mapd = @mapd AND masp = @masp;


    -- Tính tổng số lượng nhập của sản phẩm trong bảng CTPN
    SELECT @tongsoluongnhap = SUM(soluongdat)
    FROM CHI_TIET_PHIEU_NHAP ctpn, CHI_TIET_PHIEU_DAT
    WHERE mapd = @mapd AND ctpn.MASP = @masp
    GROUP BY mapd, ctpn.MASP;


    -- Kiểm tra điều kiện, nếu soluongdat < tongsoluongnhap thì không cho phép thay đổi
    IF @soluongdat < @tongsoluongnhap
    BEGIN
        PRINT N'Không thể thay đổi! Số lượng đã đặt nhỏ hơn tổng số lượng nhập.';
        ROLLBACK;  -- Hủy bỏ thao tác nếu điều kiện không thỏa mãn
    END
END



CREATE TRIGGER trg_CheckTongSoLuongNhap
ON CHI_TIET_PHIEU_NHAP
AFTER INSERT, UPDATE
AS
BEGIN
    DECLARE @masp CHAR(10);
    DECLARE @SLNHAP INT;
    DECLARE @SLDAT INT;
    DECLARE @tongsoluongnhap INT;
    DECLARE @mapd CHAR(20);

    -- Lấy dữ liệu từ bảng INSERTED
    SELECT @masp = masp, @SLNHAP = soluongnhap
    FROM INSERTED;

    -- Lấy mã phiếu đặt (mapd) từ bảng CTPD thông qua masp
    SELECT @mapd = mapd
    FROM CHI_TIET_PHIEU_DAT
    WHERE masp = @masp;

    -- Tính tổng số lượng nhập của sản phẩm trong bảng CTPN
    SELECT @tongsoluongnhap = SUM(soluongnhap) 
    FROM CHI_TIET_PHIEU_NHAP 
    WHERE masp = @masp 
    GROUP BY masp;

    -- Lấy số lượng đã đặt cho sản phẩm từ bảng CTPD
    SELECT @SLDAT = soluongdat
    FROM CHI_TIET_PHIEU_DAT
    WHERE mapd = @mapd AND masp = @masp;

    -- Kiểm tra điều kiện
    IF @tongsoluongnhap > @SLDAT
    BEGIN
        PRINT N'Tổng số lượng nhập vượt quá số lượng đã đặt!';
        ROLLBACK;
    END
END



CREATE TRIGGER CAPNHATSLKHINHAP
ON CHI_TIET_PHIEU_NHAP
AFTER INSERT, UPDATE, DELETE
AS
BEGIN
    -- Khi thêm hoặc cập nhật
    IF EXISTS (SELECT 1 FROM INSERTED)
    BEGIN
        UPDATE LUU_TRU
        SET soluonghangton = soluonghangton + I.soluongnhap
        FROM LUU_TRU LT
        JOIN INSERTED I ON LT.makho = (SELECT makho FROM PHIEU_NHAP_HANG WHERE mapn = I.mapn)
                        AND LT.masp = I.masp;
    END

    -- Khi xóa
    IF EXISTS (SELECT 1 FROM DELETED)
    BEGIN
        UPDATE LUU_TRU
        SET soluonghangton = soluonghangton - D.soluongnhap
        FROM LUU_TRU LT
        JOIN DELETED D ON LT.makho = (SELECT makho FROM PHIEU_NHAP_HANG WHERE mapn = D.mapn)
                        AND LT.masp = D.masp;
    END
END
